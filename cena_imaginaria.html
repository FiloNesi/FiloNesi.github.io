    <!-- Application Structure Plan: The SPA is designed as an interactive, multi-section tool to guide students. It starts with an introduction and character cards for context, moves to an interactive section for setting up the scene (title, location, and a dynamic pseudoscience selector), then provides a core script-building area using an accordion for Inicio, Desarrollo, and Cierre, enhanced with "idea spark" buttons.     <!-- Visualization & Content Choices:

        - Report Info: Pseudoscience examples. -> Goal: Spark conflict. -> Presentation: Interactive Button Selector. -> Interaction: Click a button to get a conflict-starting prompt. -> Justification: More engaging than a static list and actively helps initiate the dialogue.

        - Report Info: Dialogue structure. -> Goal: Guide writing. -> Presentation: Accordion with text areas. -> Interaction: "Idea Spark" buttons inject sample sentences into the text areas. -> Justification: Directly aids students in applying complex philosophical ideas, preventing writer's block.

        - Report Info: Formatting rules. -> Goal: Format output. -> Presentation: Toggle switch. -> Interaction: Show/hide relevant formatting guides. -> Justification: Simplifies the UI by only showing relevant information.

        - Library/Method: Vanilla JavaScript for all interactions.

                    btn.classList.add('ring-2', 'ring-offset-2', 'ring-teal-500');

                });

            });

            

            const accordionHeaders = document.querySelectorAll('.accordion-header');

            accordionHeaders.forEach(header => {

                header.addEventListener('click', () => {

                    const content = header.nextElementSibling;

                    const isOpen = content.style.maxHeight;

                    

                    // Close all others

                    document.querySelectorAll('.accordion-content').forEach(c => c.style.maxHeight = null);

                    document.querySelectorAll('.accordion-header span:last-child').forEach(arrow => arrow.textContent = '▼');


                    if (!isOpen) {

                        content.style.maxHeight = content.scrollHeight + "px";

                        header.querySelector('span:last-child').textContent = '▲';

                    }

                });

            });


            const ideaSparkBtns = document.querySelectorAll('.idea-spark-btn');

            ideaSparkBtns.forEach(btn => {

                btn.addEventListener('click', () => {

                    const targetId = btn.dataset.target;

                    const idea = btn.dataset.idea;

                    const textarea = document.getElementById(targetId);

                    textarea.value += (textarea.value ? '\n' : '') + idea;

                });

            const resultadoSection = document.getElementById('resultado');

            generateBtn.addEventListener('click', () => {

                const titulo = document.getElementById('titulo').value;

                const lugar = document.getElementById('lugar').value;

                const pseudocienciaSeleccionada = document.querySelector('.pseudoscience-btn.ring-2');
